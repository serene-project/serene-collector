25      @Override
26      public final void publish(final MonitoringMessageDto message) {
27          // local caches to keep from accessing the DB at each iteration
28          Map<String, Server> serverCache = new HashMap<String, Server>();
29          Map<String, ServerGroup> groupCache = new HashMap<String, ServerGroup>();
30  
31          for (ProbeValueDto pvDto : message.getProbeValues()) {
32              // add value to probe
33              ProbeValue pv = new ProbeValue();
34              pv.setProbe(findProbe(pvDto, message, serverCache, groupCache));
35              pv.setValue(Long.parseLong(pvDto.getValue()));
36              pv.setDate(new Date());
37              pv.persist();
38          }
39  
40          // TODO: send values to analyzers, should be done asynchronously
41          // (message queue ?)
42      }

44      private Probe findProbe(final ProbeValueDto pvDto,
45              final MonitoringMessageDto message,
46              final Map<String, Server> serverCache,
47              final Map<String, ServerGroup> groupCache) {
48          try {
49              if (LOG.isDebugEnabled()) {
50                  LOG.debug("Looking for probe [" + pvDto.getUuid() + "]");
51              }
52              return Probe.findProbeByUuidEquals(pvDto.getUuid());
53          } catch (EmptyResultDataAccessException dataEx) {
54              // probe does not exist
55              // TODO: if server does not exist, the group, hostname, etc...
56              // should be specified
57              if (LOG.isDebugEnabled()) {
58                  LOG.debug("Probe [" + pvDto.getUuid()
59                          + "] not found, creating it.");
60              }
61              Probe probe = new Probe();
62              probe.setName(pvDto.getName());
63              probe.setServer(findServer(message, serverCache, groupCache));
64              probe.setUuid(pvDto.getUuid());
65              probe.persist();
66  
67              if (LOG.isDebugEnabled()) {
68                  LOG.debug("Probe [" + probe + "] created.");
69              }
70              return probe;
71          }
72      }

74      private Server findServer(final MonitoringMessageDto message,
75              final Map<String, Server> serverCache,
76              final Map<String, ServerGroup> groupCache) {
77          if (serverCache.containsKey(message.getUuid())) {
78              if (LOG.isDebugEnabled()) {
79                  LOG.debug("Returning server [" + message.getUuid()
80                          + "] from local cache.");
81              }
82              return serverCache.get(message.getUuid());
83          }
84          try {
85              Server server = Server.findServerByUuidEquals(message.getUuid());
86              serverCache.put(message.getUuid(), server);
87              if (LOG.isDebugEnabled()) {
88                  LOG.debug("Returning server [" + message.getUuid()
89                          + "] from DB.");
90              }
91              return server;
92          } catch (EmptyResultDataAccessException dataEx) {
93  
94              // server does not exist
95              // TODO: if server does not exist, the group, hostname, etc...
96              // should be specified
97              if (LOG.isDebugEnabled()) {
98                  LOG.debug("Server [" + message.getUuid()
99                          + "] not found, creating it.");
100             }
101             Server server = new Server();
102             server.setUuid(message.getUuid());
103             server.setHostname(message.getHostname());
104             server.setServerGroup(findServerGroup(message.getGroup(),
105                     groupCache));
106             server.persist();
107             serverCache.put(message.getUuid(), server);
108             LOG.debug("Server created.");
109             return server;
110         }
111     }

113     private ServerGroup findServerGroup(final String name,
114             final Map<String, ServerGroup> groupCache) {
115         if (groupCache.containsKey(name)) {
116             return groupCache.get(name);
117         }
118         try {
119             ServerGroup group = ServerGroup.findServerGroupsByNameEquals(name)
120                     .getSingleResult();
121             groupCache.put(name, group);
122             return group;
123         } catch (EmptyResultDataAccessException dataExGroup) {
124             // create group if it does not exist
125             ServerGroup group = new ServerGroup();
126             group.setName(name);
127             group.persist();
128             groupCache.put(name, group);
129             return group;
130         }
