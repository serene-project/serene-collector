42  	/**
43  	 * Print a {@link MonitoringMessageDto} serialized as JSON.
44  	 */
45  	@Test
46  	public final void printJson() {
47  		MonitoringMessageDto m = new MonitoringMessageDto();
48  		m.setGroup("My Group");
49  		m.setHostname("myhost");
50  		m.setUuid(UUID.randomUUID().toString());
51  		m.setProbeValues(new ArrayList<ProbeValueDto>());
52  		ProbeValueDto pv = new ProbeValueDto();
53  		pv.setName("CPU");
54  		pv.setUuid(UUID.randomUUID().toString());
55  		pv.setValue("0.7");
56  		m.getProbeValues().add(pv);
57  		pv = new ProbeValueDto();
58  		pv.setName("disk");
59  		pv.setUuid(UUID.randomUUID().toString());
60  		pv.setValue("21");
61  		m.getProbeValues().add(pv);
62  		System.out.println(m.toJson());
63  	}
64  
65  	/**
66  	 * Test deserializing a JSON {@link MonitoringMessageDto}.
67  	 */
68  	@Test
69  	public final void deserialize() {
70  		MonitoringMessageDto m = MonitoringMessageDto
71  				.fromJsonToMonitoringMessageDto("{\"" + "group\":\"My Group\","
72  						+ "\"hostname\":\"myhost\"," + "\"probeValues\":[{\""
73  						+ "name\":\"CPU\","
74  						+ "\"uuid\":\"3ab61912-761b-4e68-9c92-a61f55daa310\","
75  						+ "\"value\":\"0.7\"" + "},{\"" + "name\":\"disk\","
76  						+ "\"uuid\":\"07185ca2-954a-4a79-8b17-da44251f9751\","
77  						+ "\"value\":\"21\"" + "}],"
78  						+ "\"uuid\":\"410f8c3d-f8d2-4597-ad49-2ad23bde306f\"}");
79  		assertEquals("My Group", m.getGroup());
80  		assertEquals(2, m.getProbeValues().size());
81  		assertEquals("CPU", m.getProbeValues().get(0).getName());
82  	}
