58  	 * Null UUID should not be accepted.
59  	 */
60  	@Test
61  	public final void uuidNull() {
62  		ProbeValueDto pv = new ProbeValueDto();
63  		pv.setUuid((String) null);
64  		BeanPropertyBindingResult errors = new BeanPropertyBindingResult(pv,
65  				"propertyValue");
66  		this.validator.validate(pv, errors);
67  		List<FieldError> fieldErrors = errors.getFieldErrors("uuid");
68  		assertEquals(1, fieldErrors.size());
69  		assertEquals("probeValue.uuid.empty", fieldErrors.get(0).getCode());
70  	}
