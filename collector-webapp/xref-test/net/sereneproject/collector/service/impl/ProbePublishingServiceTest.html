54  	/**
55  	 * Test publishing new values.
56  	 * 
57  	 * This test ensures that publishing probe values where the server / probe /
58  	 * group are not known by the system works.
59  	 */
60  	@Test
61  	@Transactional
62  	public final void publishingBrandNewValue() {
63  		MonitoringMessageDto m = new MonitoringMessageDto();
64  		m.setUuid(UUID.randomUUID().toString());
65  		m.setGroup("non existing group");
66  		m.setHostname("non existing hostname");
67  		m.setProbeValues(new ArrayList<ProbeValueDto>());
68  
69  		ProbeValueDto pv = new ProbeValueDto();
70  		pv.setName("CPU");
71  		pv.setUuid(UUID.randomUUID().toString());
72  		pv.setValue("35");
73  		m.getProbeValues().add(pv);
74  
75  		pv = new ProbeValueDto();
76  		pv.setName("Disk");
77  		pv.setUuid(UUID.randomUUID().toString());
78  		pv.setValue("256");
79  		m.getProbeValues().add(pv);
80  
81  		getProbePublishingService().publish(m);
82  	}
83  
84  	private ProbePublishingService getProbePublishingService() {
85  		return this.probePublishingService;
86  	}
