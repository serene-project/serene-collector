51  @RunWith(SpringJUnit4ClassRunner.class)
52  @ContextConfiguration(locations = "classpath:/META-INF/spring/applicationContext.xml")
53  @Transactional
54  public class RrdJpaTest {
55  
56      @Autowired
57      private RrdJpaBackendFactory rrdBackend;
58  
59      @Test
60      public void testPublishingAndRetreivingData() throws IOException {
61          String uuid = UUID.randomUUID().toString();
62          Probe probe = new Probe();
63          probe.setUuid(uuid);
64          probe.setName("probe for rrd tests");
65          probe.persist();
66          RrdDef rrdDef = new RrdDef(uuid);
67          rrdDef.addArchive(AVERAGE, 0.5, 1, 600);
68          rrdDef.addArchive(AVERAGE, 0.5, 6, 700);
69          rrdDef.addArchive(AVERAGE, 0.5, 24, 775);
70          rrdDef.addArchive(AVERAGE, 0.5, 288, 797);
71          rrdDef.addDatasource("test-datasource", GAUGE, 600, Double.NaN,
72                  Double.NaN);
73          RrdDb rrdDb = new RrdDb(rrdDef, this.rrdBackend);
74  
75          long now = new Date().getTime();
76  
77          Sample sample = rrdDb.createSample(now);
78          sample.setValue("test-datasource", 0);
79          sample.update();
80  
81          sample = rrdDb.createSample(now + 60000);
82          sample.setValue("test-datasource", 1);
83          sample.update();
84  
85          sample = rrdDb.createSample(now + 120000);
86          sample.setValue("test-datasource", 2);
87          sample.update();
88  
89          Assert.assertEquals(2.0, rrdDb.getDatasource("test-datasource")
90                  .getLastValue());
91      }
