25  package net.sereneproject.collector.web;

27  import net.sereneproject.collector.dto.MonitoringMessageDto;
28  import net.sereneproject.collector.service.ProbePublishingService;
29  import net.sereneproject.collector.validation.MonitoringMessageDtoValidator;
30  import net.sereneproject.collector.web.utils.WebUtils;
31  
32  import org.apache.log4j.Logger;
33  import org.springframework.beans.factory.annotation.Autowired;
34  import org.springframework.http.HttpStatus;
35  import org.springframework.http.ResponseEntity;
36  import org.springframework.stereotype.Controller;
37  import org.springframework.validation.BeanPropertyBindingResult;
38  import org.springframework.validation.BindingResult;
39  import org.springframework.validation.ObjectError;
40  import org.springframework.web.bind.annotation.RequestBody;
41  import org.springframework.web.bind.annotation.RequestMapping;
42  import org.springframework.web.bind.annotation.RequestMethod;
43  
44  /**
45   * Controller for posting probe values.
46   * 
47   * @author gehel
48   */
49  @RequestMapping("/probe")
50  @Controller
51  public class ProbeController {
52  
53      /** Logger. */
54      private static final Logger LOG = Logger.getLogger(ProbeController.class);
55  
56      /** Service used to publish probe values. */
57      @Autowired(required = true)
58      private ProbePublishingService probePublishingService;
59  
60      /** Validator. */
61      @Autowired(required = true)
62      private MonitoringMessageDtoValidator validator;
63  
64      /**
65       * Post a set of new probe values.
66       * 
67       * @param json
68       *            the {@link MonitoringMessageDto} in JSON format
69       * @return HTTP 201 (CREATED) if successful, HTTP 500 in case of error. In
70       *         case of error an error message in JSON format will also be
71       *         returned.
72       */
73      @RequestMapping(method = RequestMethod.POST, headers = "Accept=application/json")
74      public final ResponseEntity<String> postMonitoring(
75              @RequestBody final String json) {
76          if (LOG.isDebugEnabled()) {
77              LOG.debug("Received message : " + json);
78          }
79  
80          MonitoringMessageDto message = MonitoringMessageDto
81                  .fromJsonToMonitoringMessageDto(json);
82  
83          // validation
84          BindingResult bindingResult = new BeanPropertyBindingResult(message,
85                  "message");
86          getValidator().validate(message, bindingResult);
87          if (bindingResult.hasErrors()) {
88              for (ObjectError error : bindingResult.getAllErrors()) {
89                  LOG.error(error.getObjectName() + " : " + error.getCode());
90              }
91              return new ResponseEntity<String>(WebUtils.toJson(bindingResult),
92                      HttpStatus.INTERNAL_SERVER_ERROR);
93          }
94  
95          getProbePublishingService().publish(message);
96          return new ResponseEntity<String>(HttpStatus.CREATED);
97      }
98  
99      /**
100      * The publishing service.
101      * 
102      * @return
103      */
104     private ProbePublishingService getProbePublishingService() {
105         return this.probePublishingService;
106     }
107 
108     /**
109      * The validator.
110      * 
111      * @return
112      */
113     private MonitoringMessageDtoValidator getValidator() {
114         return this.validator;
115     }
116 }
