48  /**
49   * Custom controller collecting ProbeValues.
50   * 
51   * @author gehel
52   */
53  @RequestMapping("/collector")
54  @Controller
55  public class CollectorController {
56  
57      /** Logger. */
58      private static final Logger LOG = Logger
59              .getLogger(CollectorController.class);
60  
61      /** Publishing service doing the actuall collection. */
62      @Autowired(required = true)
63      private ProbePublishingService probePublishingService;
64  
65      /** Validator used to validate requests. */
66      @Autowired(required = true)
67      private MonitoringMessageDtoValidator validator;
68  
69      /**
70       * Collect probe values.
71       * 
72       * Agents can publish values for different probes using this service. They
73       * should send POST requests to this URL. The body of the POST should
74       * contain a monitoring message in JSON format.
75       * 
76       * Element not already existing will be created automatically.
77       * 
78       * TODO: better documentation of the exact mechanism
79       * 
80       * The message is a JSON serialized {@link MonitoringMessageDto}. It looks
81       * like :
82       * 
83       * <pre>
84       * {
85       *    "group":"My Group",
86       *    "hostname":"myhost",
87       *    "probeValues":[
88       *       {
89       *          "name":"CPU",
90       *          "uuid":"069ba4fb-3e1d-4baa-951c-bcd398379d34",
91       *          "value":"0.7"
92       *       },
93       *       {
94       *          "name":"disk",
95       *          "uuid":"ada1f0ec-76cf-4afd-889e-6c1c6a9ee7f3",
96       *          "value":"21"
97       *       }
98       *    ],
99       *    "uuid":"ce7f5a04-c94c-4eb3-b804-a3ad7f3e3251"
100      * }
101      * </pre>
102      * 
103      * Requests should be sent with a "Accept: application/json" header. The
104      * body should be of content type "application/json" and be encoded in
105      * UTF-8. The "Content-Type: application/json; charset=UTF-8" header is not
106      * required, but should be used anyway for clarity.
107      * 
108      * @param json
109      *            a message contining the new probe values
110      * @return HTTP status 201 (CREATED) if post is successfull, HTTP status 400
111      *         (BAD REQUEST) if the message was not valid. In case the message
112      *         is not valid, a JSON response containing the error's detail is
113      *         returned
114      */
115     @RequestMapping(method = RequestMethod.POST, headers = "Accept=application/json")
116     public final ResponseEntity<String> postMonitoring(
117             @RequestBody final String json) {
118         if (LOG.isDebugEnabled()) {
119             LOG.debug("Received message : [" + json + "]");
120         }
121         MonitoringMessageDto message = MonitoringMessageDto
122                 .fromJsonToMonitoringMessageDto(json);
123         BindingResult bindingResult = new BeanPropertyBindingResult(message,
124                 "message");
125         getValidator().validate(message, bindingResult);
126         if (bindingResult.hasErrors()) {
127             for (ObjectError error : bindingResult.getAllErrors()) {
128                 LOG.error(error.getObjectName() + " : " + error.getCode());
129             }
130             return new ResponseEntity<String>(WebUtils.toJson(bindingResult),
131                     HttpStatus.BAD_REQUEST);
132         }
133         getProbePublishingService().publish(message);
134         return new ResponseEntity<String>(HttpStatus.CREATED);
135     }
136 
137     /**
138      * Probe publishing service.
139      * 
140      * @return the service
141      */
142     private ProbePublishingService getProbePublishingService() {
143         return this.probePublishingService;
144     }
145 
146     /**
147      * Monitoring message validator.
148      * 
149      * @return the validator
150      */
151     private MonitoringMessageDtoValidator getValidator() {
152         return this.validator;
153     }
