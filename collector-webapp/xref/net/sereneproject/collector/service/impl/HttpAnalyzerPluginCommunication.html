57  /**
58   * A service to communicate with plugins.
59   * 
60   * This service does no treatment of the request or the response and cares only
61   * about the communication itself.
62   * 
63   * @author gehel
64   */
65  @Service
66  public class HttpAnalyzerPluginCommunication implements
67          AnalyzerPluginCommunicationService {
68  
69      /** Accept header sent to the plugin. */
70      private static final Header ACCEPT_HEADER = new BasicHeader("Accept",
71              "application/json");
72  
73      /** Content-type header sent to the plugin. */
74      private static final Header CONTENT_TYPE_HEADER = new BasicHeader("Accept",
75              "application/json");
76  
77      /** Logger. */
78      private static final Logger LOG = Logger
79              .getLogger(HttpAnalyzerPluginCommunication.class);
80  
81      /** The HTTP client used for communication. */
82      private final HttpClient client;
83  
84      /**
85       * Construct the service.
86       * 
87       * @param client
88       *            the HTTP client used for communication
89       */
90      @Autowired
91      public HttpAnalyzerPluginCommunication(final HttpClient client) {
92          this.client = client;
93      }
94  
95      /**
96       * Implementation of
97       * {@link AnalyzerPluginCommunicationService#send(URI, AnalyzerRequestDto)}.
98       * 
99       * @param uri
100      *            the URI of the analyzer
101      * @param request
102      *            the message to send
103      * @return the response from the analyzer
104      * @throws ClientProtocolException
105      *             in case of protocol violation
106      * @throws IOException
107      *             in case of generic communication errors
108      */
109     @Override
110     public final AnalyzerResponseDto send(final URI uri,
111             final AnalyzerRequestDto request) throws ClientProtocolException,
112             IOException {
113         if (LOG.isDebugEnabled()) {
114             LOG.debug("Sending message " + request.toJson() + " to plugin ["
115                     + uri + "]");
116         }
117 
118         HttpPut putRequest = new HttpPut(uri);
119         putRequest.setHeader(ACCEPT_HEADER);
120         putRequest.setHeader(CONTENT_TYPE_HEADER);
121 
122         try {
123             putRequest.setEntity(new StringEntity(request.toJson(),
124                     CharEncoding.UTF_8));
125         } catch (UnsupportedEncodingException e) {
126             throw new IllegalArgumentException(
127                     "UTF-8 should always be supported", e);
128         }
129 
130         HttpResponse httpResponse = getClient().execute(putRequest);
131 
132         // TODO: check HTTP status
133         if (httpResponse.getStatusLine().getStatusCode() != HttpStatus.SC_OK) {
134             // TODO: throw some kind of exception
135         }
136 
137         if (httpResponse.getEntity() == null) {
138             // TODO: throw some kind of exception
139         }
140 
141         Reader responseStream = new InputStreamReader(httpResponse.getEntity()
142                 .getContent(), CharEncoding.UTF_8);
143         try {
144             String jsonResponse = CharStreams.toString(responseStream);
145             return AnalyzerResponseDto
146                     .fromJsonToAnalyzerResponseDto(jsonResponse);
147         } catch (IOException ioe) {
148             throw ioe;
149         } catch (RuntimeException re) {
150             putRequest.abort();
151             throw re;
152         } finally {
153             Closeables.closeQuietly(responseStream);
154         }
155     }
156 
157     /**
158      * The HTTP client used for communication.
159      * 
160      * @return the HTTP client
161      */
162     private HttpClient getClient() {
163         return this.client;
164     }
