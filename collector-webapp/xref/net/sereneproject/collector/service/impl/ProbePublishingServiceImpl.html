52  /**
53   * Service used to publish monitoring values.
54   * 
55   * @author gehel
56   */
57  @Service
58  public class ProbePublishingServiceImpl implements ProbePublishingService {
59  
60      /** Logger. */
61      private static final Logger LOG = Logger
62              .getLogger(ProbePublishingServiceImpl.class);
63  
64      /** The message queue used to publish messages to analyzers. */
65      private final MessageQueue queue;
66  
67      /**
68       * Construct the service.
69       * 
70       * @param queue
71       *            the message queue used to publish messages to analyzers.
72       */
73      @Autowired(required = true)
74      public ProbePublishingServiceImpl(final MessageQueue queue) {
75          this.queue = queue;
76      }
77  
78      /**
79       * Publish a monitoring message.
80       * 
81       * Takes care of storing values in persistent storage, creating intermedite
82       * nodes if needed. Probe values are then dispatched to a message queue for
83       * analysis.
84       * 
85       * @param message
86       *            contains data about all probe values and where to store them
87       */
88      @Override
89      public final void publish(final MonitoringMessageDto message) {
90          // local caches to keep from accessing the DB at each iteration
91          Map<String, Server> serverCache = new HashMap<String, Server>();
92          Map<String, ServerGroup> groupCache = new HashMap<String, ServerGroup>();
93  
94          for (ProbeValueDto pvDto : message.getProbeValues()) {
95              // add value to probe
96              ProbeValue pv = new ProbeValue();
97              pv.setProbe(findOrCreateProbe(pvDto, message, serverCache,
98                      groupCache));
99              pv.setValue(pvDto.getValue());
100             pv.setDate(new Date());
101             pv.persist();
102             // queue messages so they are processed by the analyzers
103             ProbeValueDateDto toAnalyze = new ProbeValueDateDto();
104             toAnalyze.setProbeUUID(pvDto.getUuid());
105             toAnalyze.setDate(pv.getDate());
106             toAnalyze.setValue(pvDto.getValue());
107             MessageInput msg = new MessageInput();
108             msg.setObject(toAnalyze);
109             synchronized (getQueue()) {
110                 getQueue().send(msg);
111                 getQueue().notifyAll();
112             }
113         }
114     }

116     /**
117      * Find a {@link Probe}, create it if it doesnt exist.
118      * 
119      * @param pvDto
120      *            the probe value whose probe we are looking for
121      * @param message
122      *            the current message, used to retreive the server / server
123      *            group if needed
124      * @param serverCache
125      *            cache local to the call, if a server is found or created, it
126      *            is stored in this cache
127      * @param groupCache
128      *            cache local to the call, if a server group is found or
129      *            created, it is stored in this cache
130      * @return the {@link Probe} we are looking for, either found in the DB or
131      *         freshly created
132      */
133     private Probe findOrCreateProbe(final ProbeValueDto pvDto,
134             final MonitoringMessageDto message,
135             final Map<String, Server> serverCache,
136             final Map<String, ServerGroup> groupCache) {
137         try {
138             if (LOG.isDebugEnabled()) {
139                 LOG.debug("Looking for probe [" + pvDto.getUuid() + "]");
140             }
141             return Probe.findProbeByUuidEquals(pvDto.getUuid());
142         } catch (EmptyResultDataAccessException dataEx) {
143             // probe does not exist
144             // TODO: if server does not exist, the group, hostname, etc...
145             // should be specified
146             if (LOG.isDebugEnabled()) {
147                 LOG.debug("Probe [" + pvDto.getUuid()
148                         + "] not found, creating it.");
149             }
150             Probe probe = new Probe();
151             probe.setName(pvDto.getName());
152             probe.setServer(findOrCreateServer(message, serverCache, groupCache));
153             probe.setUuid(pvDto.getUuid());
154             probe.persist();
155 
156             if (LOG.isDebugEnabled()) {
157                 LOG.debug("Probe [" + probe + "] created.");
158             }
159             return probe;
160         }
161     }

163     /**
164      * Finds a {@link Server}, create it if it doesnt exist.
165      * 
166      * @param message
167      *            the message whose server we are looking for
168      * @param serverCache
169      *            cache local to the call, if a server is found or created, it
170      *            is stored in this cache
171      * @param groupCache
172      *            cache local to the call, if a server group is found or
173      *            created, it is stored in this cache
174      * @return the {@link Server} we are looking for, either found in the DB or
175      *         freshly created
176      */
177     private Server findOrCreateServer(final MonitoringMessageDto message,
178             final Map<String, Server> serverCache,
179             final Map<String, ServerGroup> groupCache) {
180         if (serverCache.containsKey(message.getUuid())) {
181             if (LOG.isDebugEnabled()) {
182                 LOG.debug("Returning server [" + message.getUuid()
183                         + "] from local cache.");
184             }
185             return serverCache.get(message.getUuid());
186         }
187         try {
188             Server server = Server.findServerByUuidEquals(message.getUuid());
189             serverCache.put(message.getUuid(), server);
190             if (LOG.isDebugEnabled()) {
191                 LOG.debug("Returning server [" + message.getUuid()
192                         + "] from DB.");
193             }
194             return server;
195         } catch (EmptyResultDataAccessException dataEx) {
196 
197             // server does not exist
198             // TODO: if server does not exist, the group, hostname, etc...
199             // should be specified
200             if (LOG.isDebugEnabled()) {
201                 LOG.debug("Server [" + message.getUuid()
202                         + "] not found, creating it.");
203             }
204             Server server = new Server();
205             server.setUuid(message.getUuid());
206             server.setHostname(message.getHostname());
207             server.setServerGroup(findOrCreateServerGroup(message.getGroup(),
208                     groupCache));
209             server.persist();
210             serverCache.put(message.getUuid(), server);
211             LOG.debug("Server created.");
212             return server;
213         }
214     }

216     /**
217      * Finds a {@link ServerGroup}, create it if it doesnt exist.
218      * 
219      * @param name
220      *            the name of the group
221      * @param groupCache
222      *            cache local to the call, if a server group is found or
223      *            created, it is stored in this cache
224      * @return the {@link ServerGroup} we are looking for, either found in the
225      *         DB or freshly created
226      */
227     private ServerGroup findOrCreateServerGroup(final String name,
228             final Map<String, ServerGroup> groupCache) {
229         if (groupCache.containsKey(name)) {
230             return groupCache.get(name);
231         }
232         try {
233             ServerGroup group = ServerGroup.findServerGroupsByNameEquals(name)
234                     .getSingleResult();
235             groupCache.put(name, group);
236             return group;
237         } catch (EmptyResultDataAccessException dataExGroup) {
238             // create group if it does not exist
239             ServerGroup group = new ServerGroup();
240             group.setName(name);
241             group.persist();
242             groupCache.put(name, group);
243             return group;
244         }
245     }
