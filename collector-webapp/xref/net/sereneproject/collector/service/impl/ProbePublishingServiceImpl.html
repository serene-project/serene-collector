57  /**
58   * Service used to publish monitoring values.
59   * 
60   * @author gehel
61   */
62  @Service
63  public class ProbePublishingServiceImpl implements ProbePublishingService {
64  
65      /** Logger. */
66      private static final Logger LOG = Logger
67              .getLogger(ProbePublishingServiceImpl.class);
68  
69      /** The message queue used to publish messages to analyzers. */
70      @Resource(name = "probeToAnalyzeQueue")
71      private BlockingQueue<ProbeValueDateDto> queue;
72  
73      private final RrdBackendFactory rrdBackendFactory;
74  
75      @Autowired
76      public ProbePublishingServiceImpl(final RrdBackendFactory rrdBackendFactory) {
77          this.rrdBackendFactory = rrdBackendFactory;
78      }
79  
80      /**
81       * Publish a monitoring message.
82       * 
83       * Takes care of storing values in persistent storage, creating intermedite
84       * nodes if needed. Probe values are then dispatched to a message queue for
85       * analysis.
86       * 
87       * @param message
88       *            contains data about all probe values and where to store them
89       * @throws IOException
90       */
91      @Override
92      public final void publish(final MonitoringMessageDto message)
93              throws IOException {
94  
95          for (ProbeDto probeDto : message.getProbes()) {
96              RrdDb rrdDb = findOrCreateRrdDb(probeDto);
97              Sample sample = rrdDb.createSample();
98              for (ValueDto valueDto : probeDto.getValues()) {
99                  sample.setValue(valueDto.getName(), valueDto.getValue());
100             }
101             sample.update();
102 
103             // queue messages so they are processed by the analyzers
104             // ProbeValueDateDto toAnalyze = new ProbeValueDateDto(
105             // UUID.fromString(pvDto.getUuid()), pv.getDate(),
106             // pvDto.getValue());
107             // getQueue().add(toAnalyze);
108         }
109     }
110 
111     private RrdDb findOrCreateRrdDb(ProbeDto probeDto) throws IOException {
112         Probe probe;
113         try {
114             if (LOG.isDebugEnabled()) {
115                 LOG.debug("Checking if probe [" + probeDto.getUuid()
116                         + "] exist.");
117             }
118             probe = Probe.findProbeByUuidEquals(probeDto.getUuid());
119         } catch (EmptyResultDataAccessException dataEx) {
120             // probe does not exist
121             if (LOG.isDebugEnabled()) {
122                 LOG.debug("Probe [" + probeDto.getUuid()
123                         + "] not found, creating it.");
124             }
125             probe = new Probe();
126             probe.setUuid(probeDto.getUuid());
127             probe.setName(probeDto.getName());
128             probe.persist();
129 
130             if (LOG.isDebugEnabled()) {
131                 LOG.debug("Probe [" + probe + "] created.");
132             }
133         }
134         if (probe.getRrd() == null) {
135             LOG.debug("Initializing new RrdDb");
136             RrdDef rrdDef = new RrdDef(probeDto.getUuid());
137             rrdDef.addArchive(AVERAGE, 0.5, 1, 600);
138             rrdDef.addArchive(AVERAGE, 0.5, 6, 700);
139             rrdDef.addArchive(AVERAGE, 0.5, 24, 775);
140             rrdDef.addArchive(AVERAGE, 0.5, 288, 797);
141             for (ValueDto value : probeDto.getValues()) {
142                 rrdDef.addDatasource(value.getName(), GAUGE, 600, Double.NaN,
143                         Double.NaN);
144             }
145             return new RrdDb(rrdDef, getRrrdBackendFactory());
146         }
147         return new RrdDb(probeDto.getUuid(), getRrrdBackendFactory());
148     }
149

157     }

161     }
