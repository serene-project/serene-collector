29  package net.sereneproject.collector.rrd;
30  
31  import java.io.IOException;
32  import java.util.Arrays;
33  import java.util.UUID;
34  
35  import net.sereneproject.collector.domain.Probe;
36  
37  import org.rrd4j.core.RrdByteArrayBackend;
38  import org.springframework.dao.EmptyResultDataAccessException;
39  
40  /**
41   * RRD Backend storing RRD database as an attibute of a {@link Probe}.
42   * 
43   * @author gehel
44   */
45  public class RrdJpaBackend extends RrdByteArrayBackend {
46  
47      private volatile boolean dirty = false;
48  
49      /**
50       * Construct a backend based on a {@link Probe}.
51       * 
52       * Note that the {@link Probe} needs to be constructed before the backend
53       * can be created.
54       * 
55       * <b>WARNING :</b> This class breaks the Lizkov substitution principle as
56       * it only accept a path that conform to the {@link UUID} specification.
57       * 
58       * @param path
59       *            UUID of the probe
60       */
61      public RrdJpaBackend(final String path) {
62          super(path);
63          try {
64              Probe probe = Probe.findProbeByUuidEquals(path);
65              if (probe.getRrd() != null) {
66                  this.buffer = Arrays.copyOf(probe.getRrd(),
67                          probe.getRrd().length);
68              }
69          } catch (EmptyResultDataAccessException e) {
70              throw new IllegalStateException("The probe [" + path
71                      + "] should exist before its RRD database is accessed.");
72          }
73      }
74  
75      @Override
76      protected synchronized void write(final long offset, final byte[] bytes)
77              throws IOException {
78          super.write(offset, bytes);
79          dirty = true;
80      }
81  
82      @Override
83      public void close() throws IOException {
84          if (dirty) {
85              Probe probe = Probe.findProbeByUuidEquals(getPath());
86              probe.setRrd(buffer);
87          }
88      }
89  }
