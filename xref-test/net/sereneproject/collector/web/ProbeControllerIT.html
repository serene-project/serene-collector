39  	private static final int SERVER_PORT = 9090;
40  
41  	/**
42  	 * Test that sending a new probe works.
43  	 * 
44  	 * Sends a message containing a server / group / probe that isnt known yet
45  	 * by the application.
46  	 * 
47  	 * @throws IOException in case of communication error
48  	 */
49  	@Test
50  	public final void sendProbe() throws IOException {
51  
52  		MonitoringMessageDto m = new MonitoringMessageDto();
53  		m.setUuid(UUID.randomUUID().toString());
54  		m.setGroup("non existing group");
55  		m.setHostname("non existing hostname");
56  		m.setProbeValues(new ArrayList<ProbeValueDto>());
57  
58  		ProbeValueDto pv = new ProbeValueDto();
59  		pv.setName("CPU");
60  		pv.setUuid(UUID.randomUUID().toString());
61  		pv.setValue("35");
62  		m.getProbeValues().add(pv);
63  
64  		pv = new ProbeValueDto();
65  		pv.setName("Disk");
66  		pv.setUuid(UUID.randomUUID().toString());
67  		pv.setValue("256");
68  		m.getProbeValues().add(pv);
69  
70  		HttpHost host = new HttpHost("localhost", SERVER_PORT, "http");
71  		DefaultHttpClient client = new DefaultHttpClient();
72  		client.getCredentialsProvider().setCredentials(
73  				new AuthScope(host.getHostName(), host.getPort()),
74  				new UsernamePasswordCredentials("admin", "admin"));
75  
76  		// Create AuthCache instance
77  		AuthCache authCache = new BasicAuthCache();
78  		// Generate BASIC scheme object and add it to the local auth cache
79  		BasicScheme basicAuth = new BasicScheme();
80  		authCache.put(host, basicAuth);
81  
82  		// Add AuthCache to the execution context
83  		BasicHttpContext localcontext = new BasicHttpContext();
84  		localcontext.setAttribute(ClientContext.AUTH_CACHE, authCache);
85  
86  		HttpPost post = new HttpPost(CONTEXT + "/probe");
87  		post.addHeader("Accept", "application/json");
88  		StringEntity body = new StringEntity(m.toJson(), "UTF-8");
89  		post.setEntity(body);
90  		HttpResponse response = client.execute(host, post, localcontext);
91  
92  		ByteStreams.copy(response.getEntity().getContent(), System.out);
93  
94  		Assert.assertEquals(HttpStatus.SC_CREATED, response.getStatusLine()
95  				.getStatusCode());
96  	}
